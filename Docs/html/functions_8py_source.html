24

211 def layer_integrate(upper_contour, lower_contour, axis, integrand = 'none', axis_sign = 'negative'):
212  """!Integrate between two non-trivial surfaces, 'upper_contour' and 'lower_contour'. The arrays ind_upper and ind_lower come from the function extract_surface.

218
