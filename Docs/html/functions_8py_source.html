21  """ nearest finds the index jusut before the search value. linear uses linear interpolation to find the location between grid points.
22  May give silly answers if the input_array is not monotonic in the search direction."""
23 
24  axis=0
25  monoton_test = np.diff(input_array,axis=axis)
26 
27  if np.all(monoton_test <= 0) or np.all(monoton_test >= 0):
28  pass
29  else:
30  warnings.warn("input field is not strictly monotonic in search direction. Strange results may occur.", RuntimeWarning)
31 
32  dist = (input_array - surface_value)
33 
34  indsz = np.repeat(np.arange(input_array.shape[0]-1).reshape((input_array.shape[0]-1,1,1)),input_array.shape[1],axis=1)
35  indsz = np.repeat(indsz.reshape((input_array.shape[0]-1,input_array.shape[1],1)),input_array.shape[2],axis=2)
36 
37  sign= np.sign(dist)
38  sign[sign==0] = -1 # replace zeros with -1
39  indices_min = np.where(np.diff(sign,axis=axis),indsz,0)
40  indices_min = (np.argmax(indices_min,axis=axis)) # to deal with multiple crossings
41 
42  if method =='nearest':
43  z_surface = np.take(axis_values,indices_min[:,:])
44 
45  elif method == 'linear':
46  z_nearest = np.take(axis_values,indices_min[:,:])
47 
48  indsy = np.repeat(np.arange(indices_min.shape[0]).reshape((indices_min.shape[0],1)),indices_min.shape[1],axis=1)
49  indsx = np.repeat(np.arange(indices_min.shape[1]).reshape((1,indices_min.shape[1])),indices_min.shape[0],axis=0)
50

52  nearest = input_array[indices_min[:,:],indsy,indsx]
53  below = input_array[indices_min[:,:]+1,indsy,indsx]

56

63  test2 = surface_value < nearest
64  direction[test1 == test2] = -1

67  test2 = surface_value > below

69 
70  test1 = nearest < surface_value
71  test2 = surface_value < below
72  direction[test1 == test2] = 1

77  ((z_nearest - np.take(axis_values,indices_min[:,:] + direction))/
78  (input_array[indices_min[:,:],indsy,indsx] -
79  input_array[indices_min[:,:] + direction,indsy,indsx]))*

97 def extract_on_surface(input_array,surface_locations,axis_values):
98  """!Extract the value of a 3D field on a 2D surface.
99  This function takes an 3 dimensional matrix 'input_array' and an 2 dimensional
100  matrix 'surface_locations' and returns a 2 dimensional matrix that contains the
101  values of input_array at the location specified by surface_locations along the third dimension using
102  the values for that axis contained in 'axis_values'. Linear interpolation is used to find the values.
103 
104  """
105 
106  axis_array = np.repeat(axis_values.reshape((axis_values.shape[0],1,1)),surface_locations.shape[0],axis=1)
107  axis_array = np.repeat(axis_array.reshape((axis_array.shape[0],axis_array.shape[1],1)),surface_locations.shape[1],axis=2)
108 
109  indsz = np.repeat(np.arange(input_array.shape[0]-1).reshape((input_array.shape[0]-1,1,1)),input_array.shape[1],axis=1)
110  indsz = np.repeat(indsz.reshape((input_array.shape[0]-1,input_array.shape[1],1)),input_array.shape[2],axis=2)
111 
112  dist = (axis_array - surface_locations)
113 
114  sign= np.sign(dist)
115  sign[sign==0] = -1 # replace zeros with -1
116  indices_above = np.where(np.diff(sign,axis=0),indsz,0)
117  indices_above = (np.nanmax(indices_above,axis=0)) # to deal with multiple crossings (shouldn't be needed)
118 
119  indsy = np.repeat(np.arange(indices_above.shape[0]).reshape((indices_above.shape[0],1)),indices_above.shape[1],axis=1)
120  indsx = np.repeat(np.arange(indices_above.shape[1]).reshape((1,indices_above.shape[1])),indices_above.shape[0],axis=0)
121 
122  values_above = input_array[indices_above[:,:],indsy,indsx]
123  values_below = input_array[indices_above[:,:]+1,indsy,indsx]
124 
125  axis_above = axis_array[indices_above[:,:],indsy,indsx]
126  axis_below = axis_array[indices_above[:,:]+1,indsy,indsx]
127 
128  surface_values = values_above + ((values_below - values_above)/(axis_below - axis_above))*(surface_locations - axis_above)
129  # value above + linear interpolation of final partial cell.
130 
131  return surface_values

134 def layer_integrate(upper_contour, lower_contour, axis_values, integrand = 'none'):
135  """!Integrate between two non-trivial surfaces, 'upper_contour' and 'lower_contour'.
136  At the moment this only works if all the inputs are defined at the same grid location.
137 
138  In MITgcm world, the axis_values needs to be Z - the tracer levels.
139 
140  The input array 'integrand' is optional. If it is not included then the output is the volume per unit area (difference in depth) between the two surfaces at each grid point,
141  """
142 
143  total = np.zeros((upper_contour.shape))
144 
145 
146  if integrand == 'none':
147  total = np.absolute(upper_contour - lower_contour)
148 
149  else:
150  axis_array = np.repeat(axis_values.reshape((axis_values.shape[0],1,1)),upper_contour.shape[0],axis=1)
151  axis_array = np.repeat(axis_array.reshape((axis_array.shape[0],axis_array.shape[1],1)),upper_contour.shape[1],axis=2)
152 
153  indsz = np.repeat(np.arange(integrand.shape[0]-1).reshape((integrand.shape[0]-1,1,1)),integrand.shape[1],axis=1)
154  indsz = np.repeat(indsz.reshape((integrand.shape[0]-1,integrand.shape[1],1)),integrand.shape[2],axis=2)
155 
156  dist = (axis_array - upper_contour)
157  sign= np.sign(dist)
158  sign[sign==0] = -1 # replace zeros with -1
159  indices_upper = np.where(np.diff(sign,axis=0),indsz,0)
160  indices_upper = (np.nanmax(indices_upper,axis=0)) # to flatten array and deal with multiple crossings
161 
162  dist = (axis_array - lower_contour)
163  sign= np.sign(dist)
164  sign[sign==0] = -1 # replace zeros with -1
165  indices_lower = np.where(np.diff(sign,axis=0),indsz,0)
166  indices_lower = (np.nanmax(indices_lower,axis=0)) # to flatten array and deal with multiple crossings
167 
168  values_upper = mitgcm.functions.extract_on_surface(integrand,upper_contour,axis_values)
169  values_lower = mitgcm.functions.extract_on_surface(integrand,lower_contour,axis_values)
170 
171  for j in xrange(0,upper_contour.shape[0]):
172  for i in xrange(0,upper_contour.shape[1]):
173  if (values_upper[j,i] is not np.ma.masked) and (values_lower[j,i] is not np.ma.masked):
174  if indices_upper[j,i] == indices_lower[j,i]:
175  # in the same cell. find midvalue and mulitply by thickness
176  total[j,i] = (values_upper[j,i] + values_lower[j,i])*(upper_contour[j,i] - lower_contour[j,i])/2
177  else:
178  # not in the same cell. Have at least an upper and lower partial cell to compute
179  upper_partial = ((values_upper[j,i] + integrand[indices_upper[j,i]+1,j,i])*
180  (upper_contour[j,i] - axis_values[indices_upper[j,i]+1]))/2
181  lower_partial = ((values_lower[j,i] + integrand[indices_lower[j,i],j,i])*
182  (axis_values[indices_lower[j,i]] - lower_contour[j,i]))/2
183 
184  total[j,i] = upper_partial + lower_partial
185 
186  if indices_lower[j,i] - indices_upper[j,i] > 1:
187  # have at least one whole cell between them (the same cell case has already been captured)
188  for k in xrange(indices_upper[j,i]+1,indices_lower[j,i]):
189  total[j,i] += (integrand[k,j,i] + integrand[k+1,j,i])*(axis_values[k] - axis_values[k+1])/2 #
190 
191  return total

210 def interp_field(field,old_x,old_y,new_x,new_y,interp_order):

212

214  ##Parameters##

216  * old_x, old_y - the axis on which the original field is defined.

228
